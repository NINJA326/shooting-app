function doGet(e) {
  const params = e && e.parameter ? e.parameter : {};
  const page = normalizeText_(params.page);
  const action = normalizeText_(params.action);
  const callback = normalizeText_(params.callback);

  if (page === 'growth') {
    return HtmlService
      .createHtmlOutputFromFile('growth')
      .setTitle('NINJA PLAYER DATA')
      .addMetaTag(
        'viewport',
        'width=device-width, initial-scale=1, maximum-scale=1, user-scalable=no'
      )
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  }

  if (action) {
    const result = handleGrowthApi_(action, params);

    if (callback) {
      return ContentService
        .createTextOutput(
          callback + '(' + JSON.stringify(result) + ')'
        )
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }

    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService
    .createTextOutput(
      JSON.stringify({
        status: 'ok',
        app: CONFIG.APP_NAME
      })
    )
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return textResponse_('no body');
    }

    const body = JSON.parse(e.postData.contents);

    (body.events || []).forEach(function(event) {
      if (
        event.type === 'message' &&
        event.message &&
        event.message.type === 'text'
      ) {
        handleTextMessage_(event);
      }
    });

    return textResponse_('ok');

  } catch (error) {
    console.error(error.stack || error);
    return textResponse_('error');
  }
}

function handleTextMessage_(event) {
  const userId = event.source && event.source.userId;
  const replyToken = event.replyToken;
  const text = normalizeText_(event.message.text);

  if (!userId || !replyToken) {
    return;
  }

  if (text === '最初から' || text === 'リセット') {
    clearSession_(userId);

    const player = getLinkedPlayer_(userId);

    if (player) {
      saveSession_(userId, {
        state: STATE.MENU,
        player: player
      });

      replyMenu_(replyToken, player);

    } else {
      startLogin_(replyToken, userId);
    }

    return;
  }

  if (text === 'キャンセル') {
    const player = getLinkedPlayer_(userId);

    clearSession_(userId);

    if (player) {
      saveSession_(userId, {
        state: STATE.MENU,
        player: player
      });

      replyMenu_(replyToken, player);

    } else {
      startLogin_(replyToken, userId);
    }

    return;
  }

  if (text === 'ログアウト') {
    unlinkPlayer_(userId);
    clearSession_(userId);

    saveSession_(userId, {
      state: STATE.LOGIN_ID
    });

    replyText_(
      replyToken,
      'ログアウトしました。\n選手IDを入力してください。'
    );

    return;
  }

  let session = getSession_(userId);
  const linkedPlayer = getLinkedPlayer_(userId);

  if (!linkedPlayer && (!session || !session.state)) {
    startLogin_(replyToken, userId);
    return;
  }

  if (!session || !session.state) {
    session = {
      state: STATE.MENU,
      player: linkedPlayer
    };

    saveSession_(userId, session);
  }

  if (
    linkedPlayer &&
    (
      text === 'シューティング' ||
      text === 'シューティング入力' ||
      text === 'シュート'
    )
  ) {
    startShootingInput_(replyToken, userId, linkedPlayer);
    return;
  }

  if (
    linkedPlayer &&
    (
      text === '身長・体重' ||
      text === '身長体重' ||
      text === '身体測定'
    )
  ) {
    startBodyInput_(replyToken, userId, linkedPlayer);
    return;
  }

  if (
    linkedPlayer &&
    (
      text === 'アジリティ' ||
      text === 'アジリティ入力'
    )
  ) {
    startAgilityInput_(replyToken, userId, linkedPlayer);
    return;
  }

  if (linkedPlayer && text === '準備中') {
    replyText_(replyToken, 'この機能は現在準備中です。');
    return;
  }

  switch (session.state) {
    case STATE.LOGIN_ID:
      handleLoginId_(replyToken, userId, text);
      break;

    case STATE.LOGIN_PASSWORD:
      handleLoginPassword_(replyToken, userId, text, session);
      break;

    case STATE.MENU:
      handleMenu_(replyToken, userId, text, session);
      break;

    case STATE.BODY_HEIGHT:
      handleBodyHeight_(replyToken, userId, text, session);
      break;

    case STATE.BODY_WEIGHT:
      handleBodyWeight_(replyToken, userId, text, session);
      break;

    case STATE.BODY_CONFIRM:
      handleBodyConfirm_(replyToken, userId, text, session);
      break;

    case STATE.AGILITY_EVENT:
      handleAgilityEvent_(replyToken, userId, text, session);
      break;

    case STATE.AGILITY_VALUE:
      handleAgilityValue_(replyToken, userId, text, session);
      break;

    case STATE.AGILITY_CONFIRM:
      handleAgilityConfirm_(replyToken, userId, text, session);
      break;

    case STATE.PRACTICE:
      handlePractice_(replyToken, userId, text, session);
      break;

    case STATE.SHOT_TYPE:
      handleShotType_(replyToken, userId, text, session);
      break;

    case STATE.POSITION:
      handlePosition_(replyToken, userId, text, session);
      break;

    case STATE.ATTEMPTS:
      handleAttempts_(replyToken, userId, text, session);
      break;

    case STATE.MADE:
      handleMade_(replyToken, userId, text, session);
      break;

    case STATE.CONFIRM:
      handleConfirm_(replyToken, userId, text, session);
      break;

    default:
      clearSession_(userId);

      if (linkedPlayer) {
        saveSession_(userId, {
          state: STATE.MENU,
          player: linkedPlayer
        });

        replyMenu_(replyToken, linkedPlayer);

      } else {
        startLogin_(replyToken, userId);
      }
  }
}

function startLogin_(replyToken, userId) {
  saveSession_(userId, {
    state: STATE.LOGIN_ID
  });

  replyText_(
    replyToken,
    '初回ログインを行います。\n' +
    'playersシートに登録されている「選手ID」を入力してください。'
  );
}

function handleLoginId_(replyToken, userId, text) {
  const player = findPlayerById_(text);

  if (!player) {
    replyText_(
      replyToken,
      '選手IDが見つかりません。\n' +
      '入力内容を確認して、もう一度入力してください。'
    );

    return;
  }

  saveSession_(userId, {
    state: STATE.LOGIN_PASSWORD,
    pendingPlayerId: player.playerId
  });

  replyText_(replyToken, 'パスワードを入力してください。');
}

function handleLoginPassword_(replyToken, userId, text, session) {
  const player = findPlayerById_(session.pendingPlayerId);

  if (!player) {
    clearSession_(userId);
    startLogin_(replyToken, userId);
    return;
  }

  if (String(player.password) !== String(text)) {
    replyText_(
      replyToken,
      'パスワードが違います。\n' +
      'もう一度入力してください。\n\n' +
      '選手IDからやり直す場合は「最初から」と送信してください。'
    );

    return;
  }

  linkPlayer_(userId, player.playerId);

  const linked = {
    playerId: player.playerId,
    playerName: player.playerName,
    category: player.category
  };

  saveSession_(userId, {
    state: STATE.MENU,
    player: linked
  });

  replyQuick_(
    replyToken,
    'ログインできました。\n' +
    linked.playerName + 'さん\n' +
    'カテゴリー：' + linked.category +
    '\n\n入力する項目を選んでください。',
    [
      'シューティング',
      '身長・体重',
      'アジリティ',
      'ログアウト'
    ]
  );
}

function handleMenu_(replyToken, userId, text, session) {
  const player = session.player || getLinkedPlayer_(userId);

  if (
    text === 'シューティング' ||
    text === 'シューティング入力' ||
    text === 'シュート'
  ) {
    startShootingInput_(replyToken, userId, player);

    text === '身長・体重' ||
    text === '身長体重' ||
    text === '身体測定'

    text === 'アジリティ' ||
    text === 'アジリティ入力'
  ) {
    startAgilityInput_(replyToken, userId, player);
    return;
  }

  replyMenu_(replyToken, player);
}

    ]

}

function handleAgilityEvent_(replyToken, userId, text, session) {
  const eventConfig = AGILITY_EVENTS[text];

  if (!eventConfig) {
    replyQuick_(
      replyToken,
      '測定種目を選択してください。',
      [
        '20mシャトルラン',
        '反復横跳び',
        'L字コーン',
        '垂直跳び',
        '120回ハンドリング',
        'キャンセル'
      ]
    );

    return;
  }

  session.agilityRecord = {
    eventName: text,
    unit: eventConfig.unit
  };

  session.state = STATE.AGILITY_VALUE;

  saveSession_(userId, session);

  replyText_(
    replyToken,
    text + 'の記録を入力してください。\n\n' +
    '例：' + eventConfig.example + '\n' +
    '単位：' + eventConfig.unit + '\n\n' +
    '数字だけ入力してください。'
  );
}

function handleAgilityValue_(replyToken, userId, text, session) {
  const eventName = session.agilityRecord.eventName;
  const eventConfig = AGILITY_EVENTS[eventName];
  const value = parseDecimal_(text);

  if (!eventConfig) {
    startAgilityInput_(replyToken, userId, session.player);
    return;
  }

  if (
    value === null ||
    value < eventConfig.min ||
    value > eventConfig.max
  ) {
    replyText_(
      replyToken,
      eventName + 'は' +
      eventConfig.min + '〜' +
      eventConfig.max + eventConfig.unit +
      'の範囲で入力してください。\n' +
      '例：' + eventConfig.example
    );

    return;
  }

  const normalizedValue = roundByDigits_(
    value,
    eventConfig.decimals
  );

  session.agilityRecord.value = normalizedValue;
  session.state = STATE.AGILITY_CONFIRM;

  saveSession_(userId, session);

  const monthKey = getCurrentMonthKey_();

  const message =
    '入力内容を確認してください。\n\n' +
    '選手：' + session.player.playerName + '\n' +
    'カテゴリー：' + session.player.category + '\n' +
    '記録月：' + monthKey + '\n' +
    '種目：' + eventName + '\n' +
    '記録：' + formatByDigits_(
      normalizedValue,
      eventConfig.decimals
    ) + eventConfig.unit + '\n\n' +
    '同じ月に記録がある場合は、今回の数値に更新されます。\n\n' +
    'この内容で保存しますか？';

  replyQuick_(
    replyToken,
    message,
    [
      '保存',
      'やり直す',
      'キャンセル'
    ]
  );
}

      'キャンセル'
    ]
  );
}

function saveAgilityRecord_(player, agilityRecord) {
  const sheet = ensureAgilitySheet_();
  const eventConfig = AGILITY_EVENTS[agilityRecord.eventName];

  if (!eventConfig) {
    throw new Error('アジリティ種目が正しくありません。');
  }

  const monthKey = getCurrentMonthKey_();
  const value = Number(agilityRecord.value);

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const monthColumn = findOrCreateMonthlyColumn_(
      sheet,
      monthKey,
      4
    );

    const row = findOrCreateAgilityRow_(
      sheet,
      player.playerName,
      agilityRecord.eventName,
      eventConfig.unit
    );

    sheet
      .getRange(row, monthColumn)
      .setValue(value);

    SpreadsheetApp.flush();

  } finally {
    lock.releaseLock();
  }

  return {
    monthKey: monthKey,
    eventName: agilityRecord.eventName,
    value: value,
    unit: eventConfig.unit,
    decimals: eventConfig.decimals
  };
}

function ensureAgilitySheet_() {
  const ss = getSpreadsheet_();
  let sheet = ss.getSheetByName(CONFIG.AGILITY_SHEET);

  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.AGILITY_SHEET);
  }

  const headers = sheet
    .getRange(1, 1, 1, 3)
    .getDisplayValues()[0]
    .map(normalizeText_);

  if (
    headers[0] !== '選手名' ||
    headers[1] !== '種目' ||
    headers[2] !== '単位'
  ) {
    sheet
      .getRange(1, 1, 1, 3)
      .setValues([
        [
          '選手名',
          '種目',
          '単位'
        ]
      ]);
  }

  return sheet;
}

function findOrCreateAgilityRow_(sheet, playerName, eventName, unit) {
  const lastRow = sheet.getLastRow();

  if (lastRow >= 2) {
    const values = sheet
      .getRange(2, 1, lastRow - 1, 3)
      .getDisplayValues();

    const targetPlayer = normalizePlayerName_(playerName);
    const targetEvent = normalizeText_(eventName);

    for (let i = 0; i < values.length; i++) {
      const rowPlayer = normalizePlayerName_(values[i][0]);
      const rowEvent = normalizeText_(values[i][1]);

      if (
        rowPlayer === targetPlayer &&
        rowEvent === targetEvent
      ) {
        return i + 2;
      }
    }
  }

  const newRow = Math.max(lastRow + 1, 2);

  sheet
    .getRange(newRow, 1, 1, 3)
    .setValues([
      [
        playerName,
        eventName,
        unit
      ]
    ]);

  return newRow;
}

function startBodyInput_(replyToken, userId, player) {
  if (!player) {
    startLogin_(replyToken, userId);
    return;
  }

  saveSession_(userId, {
    state: STATE.BODY_HEIGHT,
    player: player,
    bodyRecord: {}

  replyText_(

    '数字だけ入力してください。'
  );
}

function handleBodyHeight_(replyToken, userId, text, session) {
  const height = parseDecimal_(text);

  if (
    height === null ||
    height < 100 ||
    height > 250
  ) {
    replyText_(
      replyToken,
      '身長は100〜250cmの範囲で入力してください。\n' +
      '例：158.4'
    );

    return;
  }

  session.bodyRecord = session.bodyRecord || {};
  session.bodyRecord.height = roundOneDecimal_(height);
  session.state = STATE.BODY_WEIGHT;

  saveSession_(userId, session);

  replyText_(
    replyToken,
    '身長：' + formatNumber_(session.bodyRecord.height) + 'cm\n\n' +
    '現在の体重をkgで入力してください。\n' +
    '例：47.2\n\n' +
    '数字だけ入力してください。'
  );
}

function saveBodyRecord_(player, bodyRecord) {
  const sheet = getSpreadsheet_()
    .getSheetByName(CONFIG.BODY_SHEET);

  if (!sheet) {
    throw new Error('身体測定 シートが見つかりません。');
  }

  const height = Number(bodyRecord.height);
  const weight = Number(bodyRecord.weight);
  const monthKey = getCurrentMonthKey_();

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const monthColumn = findOrCreateMonthlyColumn_(
      sheet,
      monthKey,
      3
    );

    const heightRow = findOrCreateBodyMetricRow_(
      sheet,
      player.playerName,
      '身長'
    );

    const weightRow = findOrCreateBodyMetricRow_(
      sheet,
      player.playerName,
      '体重'
    );

    sheet
      .getRange(heightRow, monthColumn)
      .setValue(height);

    sheet
      .getRange(weightRow, monthColumn)
      .setValue(weight);

    SpreadsheetApp.flush();

  } finally {
    lock.releaseLock();
  }

  return {
    monthKey: monthKey,
    height: height,
    weight: weight
  };
}

function findOrCreateBodyMetricRow_(sheet, playerName, metric) {
  const lastRow = sheet.getLastRow();

  if (lastRow >= 2) {
    const values = sheet
      .getRange(2, 1, lastRow - 1, 2)
      .getDisplayValues();

    const targetPlayer = normalizePlayerName_(playerName);
    const targetMetric = normalizeText_(metric);

    for (let i = 0; i < values.length; i++) {
      const rowPlayer = normalizePlayerName_(values[i][0]);
      const rowMetric = normalizeText_(values[i][1]);

      if (
        rowPlayer === targetPlayer &&
        rowMetric === targetMetric
      ) {
        return i + 2;
      }
    }
  }

  const newRow = Math.max(lastRow + 1, 2);

  sheet
    .getRange(newRow, 1, 1, 2)
    .setValues([
      [
        playerName,
        metric
      ]
    ]);

  return newRow;
}

function findOrCreateMonthlyColumn_(sheet, monthKey, firstMonthColumn) {
  const lastColumn = Math.max(
    sheet.getLastColumn(),
    firstMonthColumn - 1
  );

  const headers = sheet
    .getRange(1, 1, 1, lastColumn)
    .getDisplayValues()[0];

  for (
    let column = firstMonthColumn;
    column <= headers.length;
    column++
  ) {
    if (
      normalizeMonthKey_(headers[column - 1]) === monthKey
    ) {
      return column;
    }
  }

  const newColumn = Math.max(
    lastColumn + 1,
    firstMonthColumn
  );

  sheet
    .getRange(1, newColumn)
    .setValue(monthKey);

  return newColumn;
}

function getCurrentMonthKey_() {
  return Utilities.formatDate(
    new Date(),
    CONFIG.TIMEZONE,
    'yyyy/M'
  );
}

function normalizeMonthKey_(value) {
  let text = normalizeText_(value);

  if (!text) {
    return '';
  }

  text = text
    .replace(/[年月]/g, '/')
    .replace(/日/g, '')
    .replace(/-/g, '/')
    .replace(/\./g, '/')
    .replace(/\/+/g, '/')
    .replace(/^\/|\/$/g, '');

  const match = text.match(/^(\d{4})\/(\d{1,2})/);

  if (!match) {
    return text;
  }

  return Number(match[1]) + '/' + Number(match[2]);
}

function findPlayerById_(playerId) {
  const sheet = getSpreadsheet_()
    .getSheetByName(CONFIG.PLAYERS_SHEET);

  if (!sheet) {
    throw new Error('players シートが見つかりません。');
  }

  const values = sheet.getDataRange().getDisplayValues();

  if (values.length < 2) {
    return null;
  }

  const headers = values[0].map(normalizeText_);

  const idColumn = headers.indexOf('選手ID');
  const passwordColumn = headers.indexOf('パスワード');
  const nameColumn = headers.indexOf('選手名');
  const categoryColumn = headers.indexOf('カテゴリー');

  if (
    [
      idColumn,
      passwordColumn,
      nameColumn,
      categoryColumn
    ].some(function(index) {
      return index < 0;
    })
  ) {
    throw new Error(
      'players シートの必要な列が見つかりません。'
    );
  }

  const target = normalizeText_(playerId);

  for (let i = 1; i < values.length; i++) {
    if (
      normalizeText_(values[i][idColumn]) === target
    ) {
      return {
        playerId: values[i][idColumn],
        password: values[i][passwordColumn],
        playerName: values[i][nameColumn],
        category: values[i][categoryColumn]
      };
    }
  }

  return null;
}

function calculateRate_(made, attempts) {
  return Number(attempts) > 0
    ? Math.round(
        (
          Number(made) /
          Number(attempts)
        ) * 100
      )
    : 0;
}

function parseInteger_(value) {
  const text = convertFullWidthNumber_(value)
    .replace(/本/g, '')
    .trim();

  return /^\d+$/.test(text)
    ? Number(text)
    : null;
}

function parseDecimal_(value) {
  const text = convertFullWidthNumber_(value)
    .replace(/cm/gi, '')
    .replace(/kg/gi, '')
    .replace(/秒/g, '')
    .replace(/回/g, '')
    .replace(/センチ/g, '')
    .replace(/キロ/g, '')
    .replace(/，/g, '.')
    .replace(/,/g, '.')
    .trim();

  if (!/^\d+(\.\d+)?$/.test(text)) {
    return null;
  }

  const number = Number(text);

  return isFinite(number)
    ? number
    : null;
}

function convertFullWidthNumber_(value) {
  return String(value == null ? '' : value)
    .trim()
    .replace(/[０-９]/g, function(character) {
      return String.fromCharCode(
        character.charCodeAt(0) - 0xFEE0
      );
    })
    .replace(/．/g, '.');
}

function roundOneDecimal_(value) {
  return Math.round(Number(value) * 10) / 10;
}

function roundByDigits_(value, digits) {
  const factor = Math.pow(10, Number(digits) || 0);
  return Math.round(Number(value) * factor) / factor;
}

function formatNumber_(value) {
  const number = Number(value);

  if (!isFinite(number)) {
    return '';
  }

  return number % 1 === 0
    ? String(number)
    : number.toFixed(1);
}

function formatByDigits_(value, digits) {
  const number = Number(value);

  if (!isFinite(number)) {
    return '';
  }

  if (!digits) {
    return String(Math.round(number));
  }

  return number
    .toFixed(digits)
    .replace(/0+$/, '')
    .replace(/\.$/, '');
}

function normalizeText_(value) {
  return String(value == null ? '' : value).trim();
}

function normalizePlayerName_(value) {
  return normalizeText_(value)
    .replace(/[\s　]+/g, '');
}

function validateHeaders_(sheet, requiredHeaders) {
  const headers = sheet

  const missing = requiredHeaders.filter(function(header) {

  });

  if (missing.length) {
    throw new Error(
      sheet.getName() +
      ' シートに必要な列がありません：' +
      missing.join(', ')
    );
  }
}

/* =========================================================
 * summary再集計
 * ======================================================= */

function rebuildSummary_() {
  const ss = getSpreadsheet_();

  const recordsSheet = ss.getSheetByName(
    CONFIG.RECORDS_SHEET
  );

  const summarySheet = ss.getSheetByName(
    CONFIG.SUMMARY_SHEET
  );

  if (!recordsSheet) {
    throw new Error(
      'shooting_records シートが見つかりません。'
    );
  }

  if (!summarySheet) {
    throw new Error(
      'summary シートが見つかりません。'
    );
  }

  const values = recordsSheet
    .getDataRange()
    .getDisplayValues();

  const summaryLastRow = summarySheet.getLastRow();

  if (values.length <= 1) {
    if (summaryLastRow > 1) {
      summarySheet
        .getRange(
          2,
          1,
          summaryLastRow - 1,
          7
        )
        .clearContent();
    }

    return;
  }

  const headers = values[0].map(normalizeText_);

  const columns = {
    syncType: headers.indexOf('同期種別'),
    recordId: headers.indexOf('記録ID'),
    playerName: headers.indexOf('選手名'),
    category: headers.indexOf('カテゴリー'),
    shotType: headers.indexOf('種目'),
    position: headers.indexOf('ポジション'),
    made: headers.indexOf('成功数'),
    attempts: headers.indexOf('試投数'),
    deletedAt: headers.indexOf('削除日時')
  };

  Object.keys(columns).forEach(function(key) {
    if (columns[key] < 0) {
      throw new Error(
        'shooting_records に必要な列がありません：' +
        key
      );
    }
  });

  const latestByRecordId = {};

  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    const recordId = normalizeText_(
      row[columns.recordId]
    );

    if (!recordId) {
      continue;
    }

    latestByRecordId[recordId] = row;
  }

  const totals = {};

  Object.keys(latestByRecordId).forEach(function(recordId) {
    const row = latestByRecordId[recordId];

    const syncType = normalizeText_(
      row[columns.syncType]
    ).toLowerCase();

    const deletedAt = normalizeText_(
      row[columns.deletedAt]
    );

    if (
      syncType === 'delete' ||
      deletedAt
    ) {
      return;
    }

    const playerName = normalizeText_(
      row[columns.playerName]
    );

    const category = normalizeText_(
      row[columns.category]
    );

    const shotType = normalizeText_(
      row[columns.shotType]
    );

    const position = normalizeText_(
      row[columns.position]
    );

    if (
      !playerName ||
      !shotType ||
      !position
    ) {
      return;
    }

    const made = Number(
      String(row[columns.made]).replace(/,/g, '')
    ) || 0;

    const attempts = Number(
      String(row[columns.attempts]).replace(/,/g, '')
    ) || 0;

    const key = [
      playerName,
      category,
      shotType,
      position
    ].join('|||');

    if (!totals[key]) {
      totals[key] = {
        playerName: playerName,
        category: category,
        shotType: shotType,
        position: position,
        made: 0,
        attempts: 0
      };
    }

    totals[key].made += made;
    totals[key].attempts += attempts;
  });

  const output = Object
    .keys(totals)
    .map(function(key) {
      const item = totals[key];

      const rate = item.attempts > 0
        ? Math.round(
            (
              item.made /
              item.attempts
            ) * 100
          )
        : 0;

      return [
        item.playerName,
        item.category,
        item.shotType,
        item.position,
        item.made,
        item.attempts,
        rate
      ];
    });

  output.sort(function(a, b) {
    return (
      String(a[1]).localeCompare(String(b[1]), 'ja') ||
      String(a[0]).localeCompare(String(b[0]), 'ja') ||
      String(a[2]).localeCompare(String(b[2]), 'ja') ||
      String(a[3]).localeCompare(String(b[3]), 'ja')
    );
  });

  if (summaryLastRow > 1) {
    summarySheet
      .getRange(
        2,
        1,
        summaryLastRow - 1,
        7
      )
      .clearContent();
  }

  if (output.length > 0) {
    summarySheet
      .getRange(
        2,
        1,
        output.length,
        7
      )
      .setValues(output);
  }
}
/**
 * ============================================================
 * アジリティ／アジリティ測定 安全統合
 * ============================================================
 *
 * 【目的】
 * 「アジリティ測定」のデータを「アジリティ」へ安全に統合します。
 *
 * 【安全仕様】
 * 1. 元シートは削除しません。
 * 2. 統合前に両方のバックアップシートを自動作成します。
 * 3. 統合先に既存値がある場合は上書きしません。
 * 4. 値が異なる場合は「アジリティ統合ログ」へ記録します。
 * 5. 同じ選手名＋種目を1行として月別に統合します。
 *
 * 【実行方法】
 * Apps Scriptへこのコードを追加して保存し、
 * safeMergeAgilitySheets を1回だけ実行してください。
 */

function safeMergeAgilitySheets() {
  const CONFIG = {
    TARGET_SHEET: 'アジリティ',
    SOURCE_SHEET: 'アジリティ測定',
    LOG_SHEET: 'アジリティ統合ログ',
    TIMEZONE: 'Asia/Tokyo',
    FIRST_MONTH_COLUMN: 4
  };

  const ss = getSpreadsheetForAgilityMerge_();
  const target = ss.getSheetByName(CONFIG.TARGET_SHEET);
  const source = ss.getSheetByName(CONFIG.SOURCE_SHEET);

  if (!target) {
    throw new Error('統合先「アジリティ」シートが見つかりません。');
  }

  if (!source) {
    throw new Error('統合元「アジリティ測定」シートが見つかりません。');
  }

  validateAgilityWideSheet_(target, CONFIG.TARGET_SHEET);
  validateAgilityWideSheet_(source, CONFIG.SOURCE_SHEET);

  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const timestamp = Utilities.formatDate(
      new Date(),
      CONFIG.TIMEZONE,
      'yyyyMMdd_HHmmss'
    );

    const backupTargetName = createSheetBackup_(
      ss,
      target,
      'BK_アジリティ_' + timestamp
    );

    const backupSourceName = createSheetBackup_(
      ss,
      source,
      'BK_アジリティ測定_' + timestamp
    );

    const logSheet = prepareAgilityMergeLog_(ss, CONFIG.LOG_SHEET);

    const sourceData = source.getDataRange().getDisplayValues();
    const sourceHeaders = sourceData[0];

    const monthMap = {};

    for (
      let col = CONFIG.FIRST_MONTH_COLUMN;
      col <= sourceHeaders.length;
      col++
    ) {
      const monthKey = normalizeAgilityMonthKey_(
        sourceHeaders[col - 1]
      );

      if (!monthKey) {
        continue;
      }

      monthMap[col] = findOrCreateAgilityMonthColumn_(
        target,
        monthKey,
        CONFIG.FIRST_MONTH_COLUMN
      );
    }

    const targetRowMap = buildAgilityTargetRowMap_(target);

    let copiedCount = 0;
    let skippedBlankCount = 0;
    let sameValueCount = 0;
    let conflictCount = 0;
    let createdRowCount = 0;

    const logRows = [];

    for (let row = 2; row <= sourceData.length; row++) {
      const playerName = normalizeAgilityText_(sourceData[row - 1][0]);
      const eventName = normalizeAgilityText_(sourceData[row - 1][1]);
      const unit = normalizeAgilityText_(sourceData[row - 1][2]);

      if (!playerName || !eventName) {
        continue;
      }

      const key = makeAgilityRowKey_(playerName, eventName);

      let targetRow = targetRowMap[key];

      if (!targetRow) {
        targetRow = Math.max(target.getLastRow() + 1, 2);

        target.getRange(targetRow, 1, 1, 3).setValues([
          [
            playerName,
            eventName,
            unit
          ]
        ]);

        targetRowMap[key] = targetRow;
        createdRowCount++;
      } else {
        const existingUnit = normalizeAgilityText_(
          target.getRange(targetRow, 3).getDisplayValue()
        );

        if (!existingUnit && unit) {
          target.getRange(targetRow, 3).setValue(unit);
        } else if (
          existingUnit &&
          unit &&
          existingUnit !== unit
        ) {
          conflictCount++;

          logRows.push([
            new Date(),
            playerName,
            eventName,
            '',
            '単位不一致',
            existingUnit,
            unit,
            '統合先を保持'
          ]);
        }
      }

      Object.keys(monthMap).forEach(function(sourceColumnText) {
        const sourceColumn = Number(sourceColumnText);
        const targetColumn = monthMap[sourceColumn];
        const sourceValue = normalizeAgilityText_(
          sourceData[row - 1][sourceColumn - 1]
        );

        if (!sourceValue) {
          skippedBlankCount++;
          return;
        }

        const targetCell = target.getRange(targetRow, targetColumn);
        const targetValue = normalizeAgilityText_(
          targetCell.getDisplayValue()
        );

        const monthKey = normalizeAgilityMonthKey_(
          target.getRange(1, targetColumn).getDisplayValue()
        );

        if (!targetValue) {
          targetCell.setValue(parseAgilityCellValue_(sourceValue));
          copiedCount++;

          logRows.push([
            new Date(),
            playerName,
            eventName,
            monthKey,
            'コピー',
            '',
            sourceValue,
            '統合先へ追加'
          ]);

          return;
        }

        if (
          normalizeComparableAgilityValue_(targetValue) ===
          normalizeComparableAgilityValue_(sourceValue)
        ) {
          sameValueCount++;
          return;
        }

        conflictCount++;

        logRows.push([
          new Date(),
          playerName,
          eventName,
          monthKey,
          '値の競合',
          targetValue,
          sourceValue,
          '統合先を保持'
        ]);
      });
    }

    if (logRows.length > 0) {
      const startRow = Math.max(logSheet.getLastRow() + 1, 2);

      logSheet
        .getRange(startRow, 1, logRows.length, logRows[0].length)
        .setValues(logRows);
    }

    SpreadsheetApp.flush();

    const result = {
      status: '完了',
      targetSheet: CONFIG.TARGET_SHEET,
      sourceSheet: CONFIG.SOURCE_SHEET,
      backupTarget: backupTargetName,
      backupSource: backupSourceName,
      createdRows: createdRowCount,
      copiedValues: copiedCount,
      sameValues: sameValueCount,
      conflicts: conflictCount,
      skippedBlankCells: skippedBlankCount,
      sourceDeleted: false
    };

    Logger.log(JSON.stringify(result, null, 2));

    SpreadsheetApp.getUi().alert(
      '安全統合が完了しました。\n\n' +
      '新規行：' + createdRowCount + '\n' +
      'コピーした値：' + copiedCount + '\n' +
      '同じ値：' + sameValueCount + '\n' +
      '競合：' + conflictCount + '\n\n' +
      '競合内容は「' + CONFIG.LOG_SHEET + '」で確認できます。\n' +
      '「' + CONFIG.SOURCE_SHEET + '」は削除していません。'
    );

    return result;

  } finally {
    lock.releaseLock();
  }
}

function validateAgilityWideSheet_(sheet, sheetName) {
  const lastColumn = Math.max(sheet.getLastColumn(), 3);

  const headers = sheet
    .getRange(1, 1, 1, lastColumn)
    .getDisplayValues()[0]
    .map(normalizeAgilityText_);

  if (
    headers[0] !== '選手名' ||
    headers[1] !== '種目'
  ) {
    throw new Error(
      '「' + sheetName + '」の形式を確認してください。\n' +
      'A1=選手名、B1=種目、C1=単位、D列以降=月別記録が必要です。'
    );
  }

  if (!headers[2]) {
    sheet.getRange(1, 3).setValue('単位');
  }
}

function createSheetBackup_(ss, sourceSheet, requestedName) {
  const safeName = makeUniqueSheetName_(ss, requestedName);
  const backup = sourceSheet.copyTo(ss);

  backup.setName(safeName);
  backup.hideSheet();

  return safeName;
}

function makeUniqueSheetName_(ss, requestedName) {
  let name = requestedName.slice(0, 95);
  let index = 1;

  while (ss.getSheetByName(name)) {
    const suffix = '_' + index;
    name = requestedName
      .slice(0, 95 - suffix.length) + suffix;
    index++;
  }

  return name;
}

function prepareAgilityMergeLog_(ss, sheetName) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  const headers = [
    '処理日時',
    '選手名',
    '種目',
    '月',
    '処理区分',
    '統合先の値',
    '統合元の値',
    '結果'
  ];

  sheet
    .getRange(1, 1, 1, headers.length)
    .setValues([headers]);

  return sheet;
}

function buildAgilityTargetRowMap_(sheet) {
  const map = {};
  const lastRow = sheet.getLastRow();

  if (lastRow < 2) {
    return map;
  }

  const values = sheet
    .getRange(2, 1, lastRow - 1, 2)
    .getDisplayValues();

  values.forEach(function(row, index) {
    const playerName = normalizeAgilityText_(row[0]);
    const eventName = normalizeAgilityText_(row[1]);

    if (!playerName || !eventName) {
      return;
    }

    const key = makeAgilityRowKey_(playerName, eventName);

    if (!map[key]) {
      map[key] = index + 2;
    }
  });

  return map;
}

function findOrCreateAgilityMonthColumn_(
  sheet,
  monthKey,
  firstMonthColumn
) {
  const lastColumn = Math.max(
    sheet.getLastColumn(),
    firstMonthColumn - 1
  );

  const headers = sheet
    .getRange(1, 1, 1, lastColumn)
    .getDisplayValues()[0];

  for (
    let column = firstMonthColumn;
    column <= headers.length;
    column++
  ) {
    if (
      normalizeAgilityMonthKey_(headers[column - 1]) === monthKey
    ) {
      return column;
    }
  }

  const newColumn = Math.max(
    lastColumn + 1,
    firstMonthColumn
  );

  sheet.getRange(1, newColumn).setValue(monthKey);

  return newColumn;
}

function normalizeAgilityMonthKey_(value) {
  let text = normalizeAgilityText_(value);

  if (!text) {
    return '';
  }

  text = text
    .replace(/[年月]/g, '/')
    .replace(/日/g, '')
    .replace(/-/g, '/')
    .replace(/\./g, '/')
    .replace(/\/+/g, '/')
    .replace(/^\/|\/$/g, '');

  const match = text.match(/^(\d{4})\/(\d{1,2})/);

  if (!match) {
    return '';
  }

  const year = Number(match[1]);
  const month = Number(match[2]);

  if (
    year < 2000 ||
    year > 2100 ||
    month < 1 ||
    month > 12
  ) {
    return '';
  }

  return year + '/' + month;
}

function normalizeAgilityText_(value) {
  return String(value == null ? '' : value).trim();
}

function normalizeAgilityPlayerName_(value) {
  return normalizeAgilityText_(value)
    .replace(/[\s　]+/g, '');
}

function makeAgilityRowKey_(playerName, eventName) {
  return [
    normalizeAgilityPlayerName_(playerName),
    normalizeAgilityText_(eventName)
  ].join('|||');
}

function normalizeComparableAgilityValue_(value) {
  const text = normalizeAgilityText_(value)
    .replace(/,/g, '')
    .replace(/回/g, '')
    .replace(/秒/g, '')
    .replace(/cm/gi, '')
    .trim();

  const number = Number(text);

  if (text !== '' && isFinite(number)) {
    return String(number);
  }

  return text;
}

function parseAgilityCellValue_(value) {
  const text = normalizeAgilityText_(value)
    .replace(/,/g, '')
    .replace(/回/g, '')
    .replace(/秒/g, '')
    .replace(/cm/gi, '')
    .trim();

  const number = Number(text);

  return text !== '' && isFinite(number)
    ? number
    : value;
}

function handleGrowthApi_(action, params) {
  try {
    if (action === 'growthSession') {

function growthSessionApi_(params) {

      name: player.playerName,

    }
  };
}

function growthBundleApi_(params) {
  const lineUserId = verifyGrowthIdToken_(

  };

function verifyGrowthIdToken_(idToken) {
  if (!idToken) {
    throw new Error(
      'LINE認証情報を取得できませんでした。'
    );
  }

  const channelId = PropertiesService
    .getScriptProperties()
    .getProperty('LINE_LOGIN_CHANNEL_ID');

  if (!channelId) {
    throw new Error(
      'LINE_LOGIN_CHANNEL_ID が設定されていません。'
    );
  }

  const response = UrlFetchApp.fetch(
    'https://api.line.me/oauth2/v2.1/verify',
    {
      method: 'post',
      contentType: 'application/x-www-form-urlencoded',
      payload: {
        id_token: idToken,
        client_id: channelId
      },
      muteHttpExceptions: true
    }
  );

  if (response.getResponseCode() !== 200) {
    console.error(
      'LINE verify failed: ' +
      response.getResponseCode() + ' ' +
      response.getContentText()
    );

    throw new Error(
      'LINE認証に失敗しました。LINEを閉じて開き直してください。'
    );
  }

  const payload = JSON.parse(
    response.getContentText()
  );

  if (!payload.sub) {
    throw new Error(
      'LINEユーザーIDを確認できませんでした。'
    );
  }

  return payload.sub;
}

function buildGrowthShootingV3_(playerName) {
  const sheet = getSpreadsheet_()
    .getSheetByName(CONFIG.RECORDS_SHEET);

  if (!sheet || sheet.getLastRow() < 2) {
    return emptyGrowthShootingV3_();
  }

  const values = sheet
    .getDataRange()
    .getDisplayValues();

  const headers = values[0].map(normalizeText_);

  const columns = {
    recordId: headers.indexOf('記録ID'),
    syncType: headers.indexOf('同期種別'),
    date: headers.indexOf('日付'),
    playerName: headers.indexOf('選手名'),
    shotType: headers.indexOf('種目'),
    position: headers.indexOf('ポジション'),
    made: headers.indexOf('成功数'),
    attempts: headers.indexOf('試投数'),
    deletedAt: headers.indexOf('削除日時')
  };

  Object.keys(columns).forEach(function(key) {
    if (columns[key] < 0) {
      throw new Error(
        'shooting_records に必要な列がありません：' +
        key
      );
    }
  });

  /*
   * 同じ記録IDに更新行・削除行がある場合は、
   * シート内で最後に現れる行を現在の状態として採用します。
   */
  const latestById = {};

  for (let i = 1; i < values.length; i++) {
    const row = values[i];
    const recordId = normalizeText_(
      row[columns.recordId]
    );

    if (recordId) {
      latestById[recordId] = row;
    }
  }

  const target = normalizePlayerName_(playerName);

  let totalMade = 0;
  let totalAttempts = 0;
  let latestDate = '';

  const byTypeMap = {};
  const byPositionMap = {};

  /*
   * 月別比較画面で使用する集計です。
   *
   * monthly:
   *   月ごとの成功数・試投数・成功率
   *
   * monthlyByType:
   *   月ごとの2P・3P・FT・マイカン等
   *
   * monthlyByPosition:
   *   月ごとのポジション別シュートチャート
   */
  const monthlyMap = {};
  const monthlyByTypeMap = {};
  const monthlyByPositionMap = {};

  Object.keys(latestById).forEach(function(id) {
    const row = latestById[id];

    if (
      normalizePlayerName_(
        row[columns.playerName]
      ) !== target
    ) {
      return;
    }

    if (
      normalizeText_(
        row[columns.syncType]
      ).toLowerCase() === 'delete' ||
      normalizeText_(row[columns.deletedAt])
    ) {
      return;
    }

    const made = Number(
      String(row[columns.made]).replace(/,/g, '')
    ) || 0;

    const attempts = Number(
      String(row[columns.attempts]).replace(/,/g, '')
    ) || 0;

    const shotType =
      normalizeText_(row[columns.shotType]) ||
      'その他';

    const position =
      normalizeText_(row[columns.position]) ||
      '未設定';

    const date =
      normalizeText_(row[columns.date]);

    const month = date
      ? normalizeGrowthShootingMonthV4_(date)
      : '';

    totalMade += made;
    totalAttempts += attempts;

    if (date && date > latestDate) {
      latestDate = date;
    }

    addGrowthShootingAggregateV4_(
      byTypeMap,
      shotType,
      {
        type: shotType
      },
      made,
      attempts
    );

    const positionKey =
      shotType + '|||' + position;

    addGrowthShootingAggregateV4_(
      byPositionMap,
      positionKey,
      {
        type: shotType,
        position: position
      },
      made,
      attempts
    );

    if (!month) {
      return;
    }

    addGrowthShootingAggregateV4_(
      monthlyMap,
      month,
      {
        month: month
      },
      made,
      attempts
    );

    if (!monthlyByTypeMap[month]) {
      monthlyByTypeMap[month] = {};
    }

    addGrowthShootingAggregateV4_(
      monthlyByTypeMap[month],
      shotType,
      {
        type: shotType
      },
      made,
      attempts
    );

    if (!monthlyByPositionMap[month]) {
      monthlyByPositionMap[month] = {};
    }

    addGrowthShootingAggregateV4_(
      monthlyByPositionMap[month],
      positionKey,
      {
        type: shotType,
        position: position
      },
      made,
      attempts
    );
  });

  const byType = finalizeGrowthShootingMapV4_(
    byTypeMap
  ).sort(function(a, b) {
    return a.type.localeCompare(b.type, 'ja');
  });

  const byPosition = finalizeGrowthShootingMapV4_(
    byPositionMap
  ).sort(function(a, b) {
    return (
      a.type.localeCompare(b.type, 'ja') ||
      b.attempts - a.attempts ||
      b.rate - a.rate
    );
  });

  const monthly = finalizeGrowthShootingMapV4_(
    monthlyMap
  ).sort(function(a, b) {
    return String(a.month).localeCompare(
      String(b.month)
    );
  });

  const monthlyByType = {};

  Object.keys(monthlyByTypeMap).forEach(function(month) {
    monthlyByType[month] =
      finalizeGrowthShootingMapV4_(
        monthlyByTypeMap[month]
      ).sort(function(a, b) {
        return a.type.localeCompare(b.type, 'ja');
      });
  });

  const monthlyByPosition = {};

  Object.keys(monthlyByPositionMap).forEach(function(month) {
    monthlyByPosition[month] =
      finalizeGrowthShootingMapV4_(
        monthlyByPositionMap[month]
      ).sort(function(a, b) {
        return (
          a.type.localeCompare(b.type, 'ja') ||
          b.attempts - a.attempts ||
          b.rate - a.rate
        );
      });
  });

  return {
    total: {
      made: totalMade,
      attempts: totalAttempts,
      rate: growthRateV3_(
        totalMade,
        totalAttempts
      ),
      latestDate: latestDate
    },
    byType: byType,
    byPosition: byPosition,
    monthly: monthly,
    monthlyByType: monthlyByType,
    monthlyByPosition: monthlyByPosition
  };
}

/*
 * シューティング集計用の共通加算処理です。
 * 同じ種目・月・ポジションを一つにまとめます。
 */
function addGrowthShootingAggregateV4_(
  map,
  key,
  base,
  made,
  attempts
) {
  if (!map[key]) {
    map[key] = Object.assign(
      {},
      base,
      {
        made: 0,
        attempts: 0
      }
    );
  }

  map[key].made += Number(made) || 0;
  map[key].attempts += Number(attempts) || 0;
}

/*
 * 集計途中のMapを、成長記録画面へ返す配列へ変換します。
 */
function finalizeGrowthShootingMapV4_(map) {
  return Object.keys(map).map(function(key) {
    const item = map[key];

    return Object.assign(
      {},
      item,
      {
        made: Number(item.made) || 0,
        attempts: Number(item.attempts) || 0,
        rate: growthRateV3_(
          item.made,
          item.attempts
        )
      }
    );
  });
}

/*
 * shooting_records の日付を YYYY/M 形式へ統一します。
 * 例：
 * 2026-07-24 → 2026/7
 * 2026/07/24 → 2026/7
 */
function normalizeGrowthShootingMonthV4_(value) {
  const text = normalizeText_(value)
    .replace(/\./g, '/')
    .replace(/-/g, '/');

  const match = text.match(
    /^(\d{4})\/(\d{1,2})/
  );

  if (!match) {
    return '';
  }

  const year = Number(match[1]);
  const month = Number(match[2]);

  if (
    year < 2000 ||
    year > 2100 ||
    month < 1 ||
    month > 12
  ) {
    return '';
  }

  return year + '/' + month;
}

function buildGrowthBodyV3_(playerName) {
  const sheet = getSpreadsheet_()
    .getSheetByName(CONFIG.BODY_SHEET);

  if (!sheet || sheet.getLastRow() < 2) {
    return {
      records: [],
      latest: null,
      first: null,
      change: {
        height: null,
        weight: null
      }
    };
  }

  const values = sheet.getDataRange()
    .getDisplayValues();

  const headers = values[0];
  const target = normalizePlayerName_(playerName);

  let heightRow = null;
  let weightRow = null;

  for (let i = 1; i < values.length; i++) {
    const row = values[i];

    if (
      normalizePlayerName_(row[0]) !== target
    ) {
      continue;
    }

    const metric = normalizeText_(row[1]);

    if (metric === '身長') {
      heightRow = row;
    }

    if (metric === '体重') {
      weightRow = row;
    }
  }

  const records = [];

  for (
    let column = 2;
    column < headers.length;
    column++
  ) {
    const month = normalizeMonthKey_(
      headers[column]
    );

    if (!month) {
      continue;
    }

    const height = heightRow
      ? growthNumberV3_(heightRow[column])
      : null;

    const weight = weightRow
      ? growthNumberV3_(weightRow[column])
      : null;

    if (height !== null || weight !== null) {
      records.push({
        month: month,
        height: height,
        weight: weight
      });
    }
  }

  records.sort(function(a, b) {
    return growthMonthKeyV3_(a.month) -
      growthMonthKeyV3_(b.month);
  });

  const first = records.length
    ? records[0]
    : null;

  const latest = records.length
    ? records[records.length - 1]
    : null;

  return {
    records: records,
    first: first,
    latest: latest,
    change: {
      height:
        first &&
        latest &&
        first.height !== null &&
        latest.height !== null
          ? roundOneDecimal_(
              latest.height - first.height
            )
          : null,
      weight:
        first &&
        latest &&
        first.weight !== null &&
        latest.weight !== null
          ? roundOneDecimal_(
              latest.weight - first.weight
            )
          : null
    }
  };
}

function buildGrowthAgilityV3_(playerName) {
  const sheet = getSpreadsheet_()
    .getSheetByName(CONFIG.AGILITY_SHEET);

  if (!sheet || sheet.getLastRow() < 2) {
    return {
      latest: [],
      records: []
    };
  }

  const values = sheet.getDataRange()
    .getDisplayValues();

  const headers = values[0];
  const target = normalizePlayerName_(playerName);
  const records = [];

  for (let i = 1; i < values.length; i++) {
    const row = values[i];

    if (
      normalizePlayerName_(row[0]) !== target
    ) {
      continue;
    }

    const eventName = normalizeText_(row[1]);
    const unit = normalizeText_(row[2]);

    for (
      let column = 3;
      column < headers.length;
      column++
    ) {
      const month = normalizeMonthKey_(
        headers[column]
      );

      const value = growthNumberV3_(
        row[column]
      );

      if (month && value !== null) {
        records.push({
          month: month,
          type: eventName,
          value: value,
          unit: unit
        });
      }
    }
  }

  records.sort(function(a, b) {
    return (
      growthMonthKeyV3_(a.month) -
      growthMonthKeyV3_(b.month) ||
      a.type.localeCompare(b.type, 'ja')
    );
  });

  const latestMap = {};

  records.forEach(function(record) {
    latestMap[record.type] = record;
  });

  return {
    latest: Object.keys(latestMap)
      .map(function(key) {
        return latestMap[key];
      })
      .sort(function(a, b) {
        return a.type.localeCompare(b.type, 'ja');
      }),
    records: records
  };
}

function growthRateV3_(made, attempts) {
  return Number(attempts) > 0
    ? Math.round(
        Number(made) /
        Number(attempts) *
        100
      )
    : 0;
}

function growthNumberV3_(value) {
  const text = String(
    value == null ? '' : value
  )
    .replace(/,/g, '')
    .trim();

  if (!text) {
    return null;
  }

  const number = Number(text);

  return isFinite(number)
    ? number
    : null;
}

function growthMonthKeyV3_(month) {
  const match = String(month).match(
    /^(\d{4})\/(\d{1,2})$/
  );

  return match
    ? Number(match[1]) * 100 +
      Number(match[2])
    : 0;
}
